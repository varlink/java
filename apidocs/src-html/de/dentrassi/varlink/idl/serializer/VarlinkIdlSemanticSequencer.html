039        @Override
040        public void sequence(ISerializationContext context, EObject semanticObject) {
041                EPackage epackage = semanticObject.eClass().getEPackage();
042                ParserRule rule = context.getParserRule();
043                Action action = context.getAssignedAction();
044                Set<Parameter> parameters = context.getEnabledBooleanParameters();
045                if (epackage == VarlinkIdlPackage.eINSTANCE)
046                        switch (semanticObject.eClass().getClassifierID()) {
047                        case VarlinkIdlPackage.BASIC_TYPE:
048                                if (rule == grammarAccess.getBasicTypeRule()) {
049                                        sequence_BasicType(context, (BasicType) semanticObject); 
050                                        return; 
051                                }
052                                else if (rule == grammarAccess.getElementTypeRule()) {
053                                        sequence_BasicType_ElementType(context, (BasicType) semanticObject); 
054                                        return; 
055                                }
056                                else break;
057                        case VarlinkIdlPackage.ENUM:
058                                if (rule == grammarAccess.getElementTypeRule()) {
059                                        sequence_ElementType_Enum(context, (de.dentrassi.varlink.idl.varlinkIdl.Enum) semanticObject); 
060                                        return; 
061                                }
062                                else if (rule == grammarAccess.getTypeAliasDefinitionRule()
063                                                || rule == grammarAccess.getEnumRule()) {
064                                        sequence_Enum(context, (de.dentrassi.varlink.idl.varlinkIdl.Enum) semanticObject); 
065                                        return; 
066                                }
067                                else break;
068                        case VarlinkIdlPackage.ERROR:
069                                sequence_Error(context, (de.dentrassi.varlink.idl.varlinkIdl.Error) semanticObject); 
070                                return; 
071                        case VarlinkIdlPackage.FIELD:
072                                sequence_Field(context, (Field) semanticObject); 
073                                return; 
074                        case VarlinkIdlPackage.INTERFACE:
075                                sequence_Interface(context, (Interface) semanticObject); 
076                                return; 
077                        case VarlinkIdlPackage.METHOD:
078                                sequence_Method(context, (Method) semanticObject); 
079                                return; 
080                        case VarlinkIdlPackage.OBJECT:
081                                sequence_Object(context, (de.dentrassi.varlink.idl.varlinkIdl.Object) semanticObject); 
082                                return; 
083                        case VarlinkIdlPackage.TYPE_ALIAS:
084                                sequence_TypeAlias(context, (TypeAlias) semanticObject); 
085                                return; 
086                        case VarlinkIdlPackage.TYPE_REFERENCE:
087                                if (rule == grammarAccess.getElementTypeRule()) {
088                                        sequence_ElementType_TypeReference(context, (TypeReference) semanticObject); 
089                                        return; 
090                                }
091                                else if (rule == grammarAccess.getTypeReferenceRule()) {
092                                        sequence_TypeReference(context, (TypeReference) semanticObject); 
093                                        return; 
094                                }
095                                else break;
096                        }
097                if (errorAcceptor != null)
098                        errorAcceptor.accept(diagnosticProvider.createInvalidContextOrTypeDiagnostic(semanticObject, context));
099        }

179        /**
180         * Contexts:
181         *     Member returns Error
182         *     Error returns Error
183         *
184         * Constraint:
185         *     (name=ID properties=Object)
186         */
187        protected void sequence_Error(ISerializationContext context, de.dentrassi.varlink.idl.varlinkIdl.Error semanticObject) {
188                if (errorAcceptor != null) {
189                        if (transientValues.isValueTransient(semanticObject, VarlinkIdlPackage.Literals.MEMBER__NAME) == ValueTransient.YES)
190                                errorAcceptor.accept(diagnosticProvider.createFeatureValueMissing(semanticObject, VarlinkIdlPackage.Literals.MEMBER__NAME));
191                        if (transientValues.isValueTransient(semanticObject, VarlinkIdlPackage.Literals.ERROR__PROPERTIES) == ValueTransient.YES)
192                                errorAcceptor.accept(diagnosticProvider.createFeatureValueMissing(semanticObject, VarlinkIdlPackage.Literals.ERROR__PROPERTIES));
193                }
194                SequenceFeeder feeder = createSequencerFeeder(context, semanticObject);
195                feeder.accept(grammarAccess.getErrorAccess().getNameIDTerminalRuleCall_1_0(), semanticObject.getName());
196                feeder.accept(grammarAccess.getErrorAccess().getPropertiesObjectParserRuleCall_2_0(), semanticObject.getProperties());
197                feeder.finish();
198        }
