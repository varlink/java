001package de.dentrassi.varlink.generator.util;
002
003public class PropertyInformation {
004    private final String type;
005    private final String name;
006    private final String summary;
007    private final String description;
008
009    public PropertyInformation(final String type, final String name, final String summary, final String description) {
010        this.type = type;
011        this.name = name;
012        this.summary = summary;
013        this.description = description;
014    }
015
016    public String getType() {
017        return this.type;
018    }
019
020    public String getName() {
021        return this.name;
022    }
023
024    public String getSummary() {
025        return this.summary;
026    }
027
028    public String getDescription() {
029        return this.description;
030    }
031
032}
